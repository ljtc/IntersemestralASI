  linewidth=1pt,

  frametitlefont=\normalfont\bfseries,
}

% Definimos el entorno con título opcional personalizado
\NewDocumentEnvironment{teorema}{O{} +b}
  {
    \refstepcounter{definicion}%
    \begin{mdframed}[style=estiloteorema, frametitle={%
      \textbf{Teorema~\thedefinicion%
      \ifstrempty{#1}{}{~(\textbf{#1})}}}]
  }
  {
    #2
    \end{mdframed}
  }

\begin{teorema}[Hola]\label{TeoIso}
Cualquier función \( f : A \to B \) se puede escribir como composición.
\end{teorema}

\begin{definicion}[Morfismo]
Una función entre conjuntos con estructura que la respeta.
\end{definicion}
